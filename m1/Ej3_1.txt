process usuario [id: 0..N]{
    Admin.Acceder();
    acceso();
    Admin.Salir();
}

Monitor Admin{
    int consLibres=5;
    cond cola;

    Procedure Acceder(){
        while (consLibres == 0){
            wait(cola);
        }
        consLibres-=1;
    }

    Procedure Salir(){
        consLibres+=1;
        signal(cola);
    }


}
